Combine & prepare data

# Combine the subjects' data and remove the no-feedback condition

all_data <- rbind(sub01,sub02,sub03,sub06,sub07,sub08,sub09,sub10,sub11,sub12,sub13,sub14,sub15,sub16,sub17)  |>filter(percents!="No") 


# Use absolute value of the symmetry index
all_data$SI_force <- abs(all_data$SI_force)

# Split by power condition
all_data_1p5 <- all_data |> filter(pow_condition == "1p5")
all_data_2p5 <- all_data |> filter(pow_condition == "2p5")

# Reshape to long format for plotting
all_data_long <- all_data |>
  pivot_longer(
    cols      = c(efficiency, FE),
    names_to  = "variable",
    values_to = "value"
  )

Descriptive statistics

# 1.5 W/kg
tab_1p5 <- all_data |>

# Dominant vs non-dominant force
all_data_DOM_UNDOM <- all_data |>
  group_by(subject) |>
  mutate(
    Force_DOM   = ifelse(DOM == "Right", RF_norm, LF_norm),
    Force_UNDOM = ifelse(DOM == "Right", LF_norm, RF_norm)
  )

all_data_DOM_UNDOM_1p5 <- all_data_DOM_UNDOM |>

  group_by(percents) |>

# This chunck is to check that the hypothesis of the statistical model are respected


diagnostic_lmm <- function(model, group = "subject") {
  
  # Linearity check
  cat("=== Linearity check ===\n")
  plot(model)
  
  # Residuals QQ-plot
  cat("=== Residuals QQ-plot ===\n")
  qqnorm(resid(model))
  qqline(resid(model))
  
  # Residuals histogram
  cat("=== Residuals histogram ===\n")
  hist(resid(model),
       main = "Histogram of Residuals",
       xlab = "Residuals")
  
  # Influence diagnostics
  cat("=== Group influence diagnostics ===\n")
  infl <- influence(model, group = group)
  plot(infl)
  
  cat("=== Cook's distances ===\n")
  print(cooks.distance(infl))
  
  # Random effects
  cat("=== Random effects ===\n")
  print(ranef(model))
  
  invisible(list(
    influence = infl,
    cooks_distance = cooks.distance(infl),
    random_effects = ranef(model)
  ))
}
